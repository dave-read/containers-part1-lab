Create containers
=================
# run api - will bind to port 5000/5000
docker run --name api -p 5000:5000 api
#Use Browser to navigate to localhost:5000/api/hello

# run ifconfig to get your IP address
docker run --name web -e "APIURL=192.168.0.121" -p 3000:3000 web
#Use Browser to navigate to localhost:3000

docker run -d --name web --link=api:api -e "APIURL=api" -p 3000:3000 web
docker logs web

docker rm -f web

# Create network and put containers on the same network
docker network create lab
docker network connect lab api
docker run -d --name web --network=lab -e "APIURL=api" -p 3000:3000 web
docker logs web
docker network inspect lab
